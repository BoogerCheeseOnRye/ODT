isValidModelPath(filePath) {
    const normalized = path.normalize(filePath);
    const appNormalized = path.normalize("E:\dashboard-appv2");
    const modelsDirNormalized = path.normalize("E:\models");
    
    // ❌ REJECT - Path inside app folder
    if (normalized.startsWith(appNormalized)) {
        console.log("BLOCKED: Path in app folder");
        return false;  // HTTP 403
    }
    
    // ✓ ACCEPT - Path inside models folder
    if (normalized.startsWith(modelsDirNormalized)) {
        console.log("ALLOWED: Path in models folder");
        return true;  // Proceed
    }
    
    // ❌ REJECT - Path neither in app nor models
    console.log("BLOCKED: Path not in models folder");
    return false;  // HTTP 403
}

Examples:
  isValidModelPath("E:\models\qwen.gguf")              → true ✓
  isValidModelPath("E:\models\subfolder\phi.gguf")     → true ✓
  isValidModelPath("E:\dashboard-appv2\models\m.gguf") → false ✗
  isValidModelPath("E:\dashboard-appv2\m.gguf")        → false ✗
  isValidModelPath("C:\other\model.gguf")              → false ✗
  isValidModelPath("E:\other\model.gguf")              → false ✗
